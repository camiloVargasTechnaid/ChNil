47 /**
48  * Configures the reference voltage used for analog input
49  * (i.e. the value used as the top of the input range). The options are:
50  * @param[in] mode the ADC reference mode.
51  - DEFAULT: the default analog reference of 5 volts (on 5V Arduino boards)
52  or 3.3 volts (on 3.3V Arduino boards)
53  - INTERNAL: an built-in reference, equal to 1.1 volts on the ATmega168 or
54  ATmega328 and 2.56 volts on the ATmega8 (not available on the Arduino Mega)
55  - INTERNAL1V1: a built-in 1.1V reference (Arduino Mega only)
56  - INTERNAL2V56: a built-in 2.56V reference (Arduino Mega only)
57  - EXTERNAL: the voltage applied to the AREF pin (0 to 5V only) is used
58  as the reference.
59 */
60 void chAnalogReference(uint8_t mode)
61 {
62  // can't actually set the register here because the default setting
63  // will connect AVCC and the AREF pin, which would cause a short if
64  // there's something connected to AREF.
65  nil_analog_reference = mode;
66 }

68 /**
69  * Read the value from the specified analog pin.
70  * @note nilAnalogRead() sleeps, if possible, while the ADC conversion
71  * is in progress.
72  * @param[in] pin the number of the analog input pin to read from (0 to 5 on
73  * most boards, 0 to 7 on the Mini and Nano, 0 to 15 on the Mega)
74  *
75  * @return pin ADC conversion value (0 to 1023).
76  *
77  */
78 int chAnalogRead(uint8_t pin) {
79 #if defined(__AVR_ATmega32U4__)
80  pin = analogPinToChannel(pin);
81  ADCSRB = (ADCSRB & ~(1 << MUX5)) | (((pin >> 3) & 0x01) << MUX5);
82 #elif defined(ADCSRB) && defined(MUX5)
83  // the MUX5 bit of ADCSRB selects whether we're reading from channels
84  // 0 to 7 (MUX5 low) or 8 to 15 (MUX5 high).
85  ADCSRB = (ADCSRB & ~(1 << MUX5)) | (((pin >> 3) & 0x01) << MUX5);
86 #endif
87 
88  // set the analog reference (high two bits of ADMUX) and select the
89  // channel (low 4 bits). this also sets ADLAR (left-adjust result)
90  // to 0 (the default).
91 #if defined(ADMUX)
92  ADMUX = (nil_analog_reference << 6) | (pin & 0x07);
93 #endif
94 
95  if (!chIsIdleThread()) {
96  // Not idle thread so use interrupt and sleep.
97  ADCSRA |= (1 << ADIE) | (1 << ADSC);
98  chSemWait(&adcSem);
99  ADCSRA &= ~(1 << ADIE);
100  } else {
101  ADCSRA |= (1 << ADSC);
102  // ADSC is cleared when the conversion finishes
103  while (ADCSRA & (1 << ADSC));
104  }
105  // this will access ADCL first.
106  return ADC;
107 }
