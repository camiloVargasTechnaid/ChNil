30 /**
31  * Start periodic timer 1 signals.
32  *
33  * @param[in] microseconds The period in microseconds between signals.
34  * @note The maximum value for microseconds is 4,194,304 for a 16 MHz CPU.
35  */
36  void chTimer1Start(uint32_t microseconds){
37  // CPU cycles per time period.
38  long cycles = (F_CPU / 1000000) * microseconds;
39 
40  chTimer1Stop();
41 
42  uint8_t tshift;
43  if (cycles <= 0X10000) {
44  // no prescale, CTC mode
45  TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS10);
46  tshift = 0;
47  } else if (cycles <= 0X10000*8L) {
48  // prescale 8, CTC mode
49  TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS11);
50  tshift = 3;
51  } else if (cycles <= 0X10000*64L) {
52  // prescale 64, CTC mode
53  TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS11) | (1 << CS10);
54  tshift = 6;
55  } else if (cycles <= 0X10000*256L) {
56  // prescale 256, CTC mode
57  TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS12);
58  tshift = 8;
59  } else {
60  // prescale 1024, CTC mode
61  TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS12) | (1 << CS10);
62  tshift = 10;
63  // microseconds too large, use max time.
64  if (cycles > 0X10000*1024L) cycles = 0X10000*1024L;
65  }
66  // divide by prescaler
67  cycles >>= tshift;
68 
69  // set TOP for timer reset
70  chSysLock();
71  ICR1 = cycles - 1;
72  TCNT1 = 0;
73  // clear pending interrupt
74  TIFR1 |= (1 << OCF1B);
75  // enable compare B interrupt
76  TIMSK1 = (1 << OCIE1B);
77  chSysUnlock();
78 
79  }
80  /** Stop timer1 signals. */
81  void chTimer1Stop() {
82  TCCR1A = 0;
83  TCCR1B = 0;
84  TIMSK1 = 0;
85  }
86 /** Sleep while waiting for timer1 signal.
87  * @note This function should not be used in the idle thread.
88  * @return true if success or false if overrun or error.
89  */
90  bool chTimer1Wait() {
91  // Idle thread can't sleep.
92  if (chIsIdleThread()) return false;
93  if (chSemWaitTimeout(&timerSem, TIME_IMMEDIATE) != MSG_TIMEOUT) return false;
